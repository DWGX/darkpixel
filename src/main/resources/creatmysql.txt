-- 创建数据库（如果不存在）
CREATE DATABASE IF NOT EXISTS darkpixel
 CHARACTER SET utf8mb4
 COLLATE utf8mb4_unicode_ci;

-- 使用 darkpixel 数据库
USE darkpixel;

-- 创建 players 表
CREATE TABLE IF NOT EXISTS players (
 uuid VARCHAR(36) PRIMARY KEY,
 name VARCHAR(16) NOT NULL,
 `rank` VARCHAR(50) DEFAULT 'member',
 score INT DEFAULT 0,
 join_particle VARCHAR(50) DEFAULT 'FIREWORK',
 join_message VARCHAR(255) DEFAULT '欢迎 {player} 加入服务器！',
 chat_color VARCHAR(50) DEFAULT 'normal',
 show_rank BOOLEAN DEFAULT FALSE,
 show_vip BOOLEAN DEFAULT FALSE,
 show_group BOOLEAN DEFAULT TRUE,
 ban_until BIGINT DEFAULT 0,
 ban_reason VARCHAR(255),
 loginCount INT DEFAULT 0,
 signInCount INT DEFAULT 0,
 x DOUBLE DEFAULT 0,
 y DOUBLE DEFAULT 0,
 z DOUBLE DEFAULT 0,
 world VARCHAR(50) DEFAULT 'world',
 lastSignIn BIGINT DEFAULT 0,
 effects_enabled BOOLEAN DEFAULT TRUE,
 particle VARCHAR(50) DEFAULT 'FIREWORK',
 INDEX idx_name (name),
 INDEX idx_rank (`rank`)
) ENGINE=InnoDB CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建 groups 表
CREATE TABLE IF NOT EXISTS `groups` (
 name VARCHAR(50) PRIMARY KEY,
 color VARCHAR(50),
 emoji VARCHAR(50),
 badge VARCHAR(50),
 prefix VARCHAR(50)
) ENGINE=InnoDB CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

 -- 创建 player_groups 表（玩家与组的关系表）
CREATE TABLE IF NOT EXISTS player_groups (
 uuid VARCHAR(36),
 player_name VARCHAR(16),
 group_name VARCHAR(50),
 PRIMARY KEY (uuid, group_name),
 FOREIGN KEY (uuid) REFERENCES players(uuid) ON DELETE CASCADE ON UPDATE CASCADE,
 FOREIGN KEY (group_name) REFERENCES `groups`(name) ON DELETE CASCADE ON UPDATE CASCADE,
 INDEX idx_uuid (uuid)
) ENGINE=InnoDB CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建 bans 表
CREATE TABLE IF NOT EXISTS bans (
 uuid VARCHAR(36) PRIMARY KEY,
 player_name VARCHAR(16),
 ip VARCHAR(50),
 client_identifier VARCHAR(255),
 ban_until BIGINT DEFAULT 0,
 ban_reason VARCHAR(255),
 INDEX idx_player_name (player_name)
) ENGINE=InnoDB CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建 blacklist 表
CREATE TABLE IF NOT EXISTS blacklist (
 identifier VARCHAR(255) PRIMARY KEY
) ENGINE=InnoDB CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
